def calculate_sum(numbers):
   
    sum = 0
    for num in numbers:
        sum += num
    return sum
def find_maximum(numbers):
  
    
    max_num = numbers[0]
    for num in numbers:
        if num > max_num:  
            max_num = num  
    return max_num
